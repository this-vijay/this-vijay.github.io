\begin{prob}
Compute the de Rham cohomology ring of the $n$-sphere $S^n$.
\end{prob}

\begin{prob}
Compute the de Rham cohomology ring of the multiply-punctured Euclidean space $\R^n \setminus \{p_1, \ldots, p_m\}$,
where $m \geq 1$.
\end{prob}

\begin{prob}
 Compute the de Rham cohomology ring of the $3$-dimensional torus $T^3 \cong \R^3 / \Z^3$.
\end{prob}

\begin{prob}
Compute the de Rham cohomology vector spaces of the real projective space $\R \P^2$.
\end{prob}

\begin{prob}
Compute the de Rham cohomology vector spaces of the
compact orientable surface of genus $3$.
\end{prob}
