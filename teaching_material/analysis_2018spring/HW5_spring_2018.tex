\begin{prob}
 Let $A$ be an open subset of $\R^n$, and let $H: A \to \R^n$ be a
 function of class $C^1$ such that $DH(a) = \+0$ for some $a \in A$.
 Prove that for any $\epsilon > 0$, there exists a neighborh
 \[
\frac{||H(x_0) - H(x_1)||}{||x_0 - x_1||} < \epsilon 
 \]
 for any pair of points $x_0$ and $x_1$ in a sufficiently
 small neighborhood of $a$.
\end{prob}
